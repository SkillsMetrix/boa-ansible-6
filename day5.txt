- name: copy index file
  template: src=index.html dest=/home/ubuntu
  register: copy_result
  ignore_errors: true

- name: print the value
  debug:
    var: copy_result

- name: failed all
  template: src=update.html dest=/home/ubuntu
  failed_when:
    - copy_result.failed == true
    - '"Could not find or access in copy_result.msg"'
  any_errors_fatal: true

- name: print another value
  debug:
    var: copy_result
